        // --- Form Submission Logic ---

        // This JavaScript will find the form, listen for a "submit" event,

        // send the data to your Make.com webhook, and THEN show the success message.


        const form = document.getElementById('request-form');

        const formContainer = document.getElementById('form-container');

        const successMessage = document.getElementById('success-message');

        

        // --- !!! PASTE YOUR MAKE.COM WEBHOOK URL HERE !!! ---

        const webhookURL = 'https://hook.us2.make.com/4jkjrcztqst43p9t5dd1p8gm93cesro8'; 

        // --- e.g., 'https://hook.eu1.make.com/a1b2c3d4e5f6g7h8' ---


        form.addEventListener('submit', function(event) {

            // Prevent the form from actually submitting (which would cause a page reload)

            event.preventDefault();


            // 1. Create a data object from the form

            // We use the 'id' of each input field

            const formData = {

                productURL: document.getElementById('product-url').value,

                productName: document.getElementById('product-name').value,

                productOptions: document.getElementById('product-options').value,

                quantity: document.getElementById('quantity').value,

                originalPrice: document.getElementById('original-price').value,

                fullName: document.getElementById('full-name').value,

                email: document.getElementById('email').value,

                deliveryAddress: document.getElementById('delivery-address').value,

                phone: document.getElementById('phone').value

            };


            // 2. Send the data to the Make.com webhook

            fetch(webhookURL, {

                method: 'POST',

                headers: {

                    'Content-Type': 'application/json'

                },

                body: JSON.stringify(formData)

            })

            .then(response => {

                if (response.ok) {

                    // 3. If the data was sent successfully, show the success message

                    // Hide the form

                    form.classList.add('hidden');

                    // Show the success message

                    successMessage.classList.remove('hidden');

                    // Optional: Scroll back to the top of the form container to show the message

                    formContainer.scrollIntoView({ behavior: 'smooth' });

                } else {

                    // Handle errors if the webhook fails

                    alert('Error: Could not submit quote. Please try again.');

                }

            })

            .catch(error => {

                // Handle network errors

                console.error('Fetch Error:', error);

                alert('Error: Could not connect. Please check your internet connection.');

            });

        });
